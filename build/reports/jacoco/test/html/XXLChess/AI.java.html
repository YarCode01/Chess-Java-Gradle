package XXLChess;

import java.util.*;

public class AI {
    private PlayerColour colour;
    private ArrayList<ChessPiece> available_pieces;
    

    public AI(App app, PlayerColour colour){
        this.colour = colour;
        this.available_pieces = app.getPieces(this.colour);
    }

    public void move(App app){
        Random rand = new Random();
        this.available_pieces = app.getPieces(this.colour);
        app.reset_last_move();
        for(int i = 0; i < 100; i++){
            int index = rand.nextInt(this.available_pieces.size());
            ChessPiece piece = this.available_pieces.get(index);
            ArrayList<int[]> available_captures = piece.getAvailableCaptures(app);
            ArrayList<int[]> illegal_moves = piece.getIllegalMoves(app);
            available_captures.removeAll(illegal_moves);
            if(available_captures.size() > 0){
                int[] available_move = available_captures.get(0);
                app.select_move(piece.getPosition()[0],piece.getPosition()[1], available_move[0], available_move[1]);
                piece.move(app, available_move[0], available_move[1]);
                return;
            }
        }
        for(int i = 0; i < 100; i++){
            int rand_index = rand.nextInt(this.available_pieces.size());
            ChessPiece piece = this.available_pieces.get(rand_index);
            ArrayList<int[]> available_moves = piece.getAvailableMoves(app);
            ArrayList<int[]> illegal_moves = piece.getIllegalMoves(app);
            available_moves.removeAll(illegal_moves);
            if(available_moves.size() > 0){
                int[] available_move = available_moves.get(0);
                app.select_move(piece.getPosition()[0],piece.getPosition()[1], available_move[0], available_move[1]);
                piece.move(app, available_move[0], available_move[1]);
                return;
            }
        }
    }
}
